Post New User

Endpoint URL: /users Method: POST

Expected Admin Input 
{
                        "first_name": "Admin",
                        "last_name": "Admin",
                        "email": "Admin",
                        "phone": "555-555-5555",
                        "password": "Admin",
                        "is_admin": 1,
                        "airport": "Denver"  
                }
Expected Non-Admin Input 
{
                        "first_name": "bar",
                        "last_name": "bar",
                        "email": "bar",
                        "phone": "555-555-5555",
                        "password": "bar",
                        "is_admin": 0,  
                }
Example Output 

                    {
                        "id": 13
                    }
